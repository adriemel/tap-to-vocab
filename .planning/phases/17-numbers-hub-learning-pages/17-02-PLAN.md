**Create /home/desire/tap-to-vocab/numbers-learn.html:**

Loads numbers-data.js, then an inline IIFE script that: (1) parses ?range param, (2) filters NUMBERS, (3) renders rows using textContent (NOT innerHTML for item data — security requirement), (4) sets btn-take-test href.

CSS for number rows is inlined in a `<style>` block in `<head>`.

```html
<!doctype html>
<html lang="en">
<head>
  <meta charset="utf-8" />
  <meta name="viewport" content="width=device-width,initial-scale=1" />
  <link rel="icon" href="data:image/svg+xml,<svg xmlns='http://www.w3.org/2000/svg' viewBox='0 0 100 100'><text y='.9em' font-size='90'>🔢</text></svg>">
  <title>Números – Tap‑to‑Vocab</title>
  <link rel="stylesheet" href="/assets/css/styles.css" />
  <style>
    #number-list {
      display: flex;
      flex-direction: column;
      gap: 0;
      margin-bottom: 24px;
    }
    .number-row {
      display: flex;
      align-items: center;
      gap: 12px;
      padding: 10px 4px;
      border-bottom: 1px solid rgba(174,182,217,0.12);
      font-size: 1.1rem;
    }
    .number-row:last-child {
      border-bottom: none;
    }
    .number-numeral {
      font-weight: 700;
      color: var(--accent);
      min-width: 36px;
      text-align: right;
    }
    .number-dash {
      color: var(--muted);
    }
    .number-word {
      color: var(--ink);
      font-size: 1.1rem;
    }
    #btn-take-test {
      display: block;
      width: 100%;
      margin-top: 8px;
      text-align: center;
    }
  </style>
</head>
<body>
  <div class="container">
    <div class="card">
      <div style="display:flex; align-items:center; justify-content:space-between; gap:10px; flex-wrap:wrap; margin-bottom:16px;">
        <h1 style="margin:0">🔢 <span id="range-title"></span></h1>
        <div style="display:flex; gap:8px;">
          <a class="btn secondary" href="/">🏠 Home</a>
          <a class="btn secondary" href="/numbers.html">← Numbers</a>
        </div>
      </div>
      <div id="number-list"></div>
      <a id="btn-take-test" class="btn" href="#">Take a Test →</a>
    </div>
  </div>
  <script src="/assets/js/numbers-data.js"></script>
  <script>
    (function () {
      var params = new URLSearchParams(location.search);
      var range = params.get("range") || "1-20";
      var parts = range.split("-");
      var lo = parseInt(parts[0], 10);
      var hi = parseInt(parts[1], 10);

      // Validate parsed integers — fall back to default if malformed
      if (isNaN(lo) || isNaN(hi) || lo < 1 || hi > 100 || lo > hi) {
        lo = 1;
        hi = 20;
        range = "1-20";
      }

      // Update page heading — use textContent, never inject range param into innerHTML
      document.getElementById("range-title").textContent = lo + " – " + hi;

      // Render number rows
      var list = window.NUMBERS.filter(function (item) {
        return item.n >= lo && item.n <= hi;
      });
      var container = document.getElementById("number-list");
      list.forEach(function (item) {
        var row = document.createElement("div");
        row.className = "number-row";

        var numEl = document.createElement("span");
        numEl.className = "number-numeral";
        numEl.textContent = item.n;

        var dashEl = document.createElement("span");
        dashEl.className = "number-dash";
        dashEl.textContent = "—";

        var wordEl = document.createElement("span");
        wordEl.className = "number-word";
        wordEl.textContent = item.es;

        row.appendChild(numEl);
        row.appendChild(dashEl);
        row.appendChild(wordEl);
        container.appendChild(row);
      });

      // Wire "Take a Test" button — passes same range param to quiz page
      document.getElementById("btn-take-test").href = "/numbers-quiz.html?range=" + range;
    })();
  </script>
</body>
</html>
```

Key security note: `item.n` and `item.es` are set via `.textContent` — never via `.innerHTML`. The `range` param is only used for parseInt filtering and the href attribute value (href is safe, not innerHTML). The heading shows `lo + " – " + hi` (computed integers), not the raw param string.
